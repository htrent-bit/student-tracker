function doGet(e) { return handleResponse(e); } function doPost(e) { return handleResponse(e); } function handleResponse(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; // Get ID from the URL (?studentID=xxx) var id = (e && e.parameter && e.parameter.studentID) ? e.parameter.studentID : "No ID"; // Append Row: [Timestamp, ID] sheet.appendRow([new Date(), id]); // Return a clear "OK" to the browser return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); }
